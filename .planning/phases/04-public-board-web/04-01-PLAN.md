   ```ts
   /**
    * Extract YouTube video_id from various URL formats.
    * Returns null for non-YouTube or malformed URLs.
    * Out of v1 scope: m.youtube.com, youtube.com/shorts (per RESEARCH Pattern 7).
    */
   export function extractYouTubeVideoId(url: string): string | null {
     const m = url.match(/(?:youtube\.com\/(?:watch\?v=|embed\/)|youtu\.be\/)([\w-]{11})/);
     return m?.[1] ?? null;
   }

   /**
    * Build YouTube watch URL with optional timestamp (per D-15).
    * If videoId can't be extracted, returns null.
    * If timestampSec is null/0/fractional<1, omits ?t=.
    */
   export function buildYouTubeWatchUrl(linkUrl: string, timestampSec: number | null): string | null {
     const videoId = extractYouTubeVideoId(linkUrl);
     if (!videoId) return null;
     const base = `https://www.youtube.com/watch?v=${videoId}`;
     const floored = timestampSec != null ? Math.floor(timestampSec) : 0;
     return floored > 0 ? `${base}&t=${floored}s` : base;
   }
   ```

   ```ts
   /**
    * Google Static Maps URL builder for OG image (per CONTEXT D-07).
    * - Single markers= param with brand-500 color (0xF97316) + size:mid.
    * - Max 10 markers (truncated). No URL signing in v1 (D-07).
    * - No center/zoom ‚Üí Google auto-fits bbox.
    */
   export interface MarkerLatLng {
     lat: number;
     lng: number;
   }

   export interface BuildStaticMapsOpts {
     places: MarkerLatLng[];
     size: { width: number; height: number };
     scale?: 1 | 2;
     apiKey: string;
     styleParams?: string[];
   }

   export function buildStaticMapsUrl(opts: BuildStaticMapsOpts): string {
     const { places, size, scale = 2, apiKey, styleParams = [] } = opts;
     if (places.length === 0) {
       throw new Error('buildStaticMapsUrl: at least 1 place required');
     }

     const truncated = places.slice(0, 10);
     const markerStr =
       `color:0xF97316|size:mid|` +
       truncated.map((p) => `${p.lat.toFixed(6)},${p.lng.toFixed(6)}`).join('|');

     const params = new URLSearchParams({
       size: `${size.width}x${size.height}`,
       scale: scale.toString(),
       maptype: 'roadmap',
       key: apiKey,
     });
     params.append('markers', markerStr);

     for (const s of styleParams) {
       params.append('style', s);
     }

     return `https://maps.googleapis.com/maps/api/staticmap?${params.toString()}`;
   }

    * Removes POI labels, simplifies road labels, desaturates geometry.
   ```
