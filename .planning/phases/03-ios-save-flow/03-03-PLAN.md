```typescript
// ----- resolve-place Edge Function contract (Phase 3 SAVE-05) -----

/**
 * Request to the resolve-place Edge Function. Either `query` (free text)
 * or `{lat,lng}` (location bias) must be provided. `query` is the primary
 * mode per D-07 (text input + dropdown); lat/lng bias is for future
 * map long-press (deferred to v2).
 */
export const ResolvePlaceRequestSchema = z
  .object({
    query: z.string().min(1).max(200).optional(),
    lat: z.number().min(-90).max(90).optional(),
    lng: z.number().min(-180).max(180).optional(),
    language: z.string().min(2).max(8).default('ko'),
  })
  .refine((v) => v.query !== undefined || (v.lat !== undefined && v.lng !== undefined), {
    message: 'Either query or (lat,lng) is required',
  });

export type ResolvePlaceRequest = z.infer<typeof ResolvePlaceRequestSchema>;

/**
 * One resolved place candidate. Shape mirrors Google Places API v1
 * Text Search response with our FieldMask whitelist.
 */
export const ResolvedPlaceSchema = z.object({
  google_place_id: z.string().min(1),
  displayName: z.string(),
  formattedAddress: z.string().nullable(),
  location: z.object({
    lat: z.number(),
    lng: z.number(),
  }),
  primaryType: z.string().nullable(),
});

export type ResolvedPlace = z.infer<typeof ResolvedPlaceSchema>;

export const ResolvePlaceResponseSchema = z.object({
  places: z.array(ResolvedPlaceSchema).max(5), // D-07: max 5
});

export type ResolvePlaceResponse = z.infer<typeof ResolvePlaceResponseSchema>;
```

**A. Create `supabase/functions/resolve-place/pipeline/places-search.ts`:**

```typescript
// resolve-place pipeline: Google Places API v1 Text Search with explicit FieldMask.
//
// Cloned from supabase/functions/extract-youtube/pipeline/places.ts but generalized:
// - maxResultCount = 5 (D-07: dropdown max 5 results)
// - returns array of normalized ResolvedPlace (not single)
// - optional locationBias from lat/lng (50km radius)
// - includes primaryType in mask (manual pin UI may show it in the future)

export const FIELD_MASK = [
  'places.id',
  'places.displayName',
  'places.formattedAddress',
  'places.location',
  'places.primaryType',
].join(',');

export interface ResolvedPlace {
  google_place_id: string;
  displayName: string;
  formattedAddress: string | null;
  location: { lat: number; lng: number };
  primaryType: string | null;
}

export interface SearchOptions {
  query?: string;
  lat?: number;
  lng?: number;
  language: string;
}

export async function searchPlaces(opts: SearchOptions): Promise<ResolvedPlace[]> {
  const apiKey = Deno.env.get('GOOGLE_PLACES_SERVER_KEY');
  if (!apiKey) throw new Error('GOOGLE_PLACES_SERVER_KEY missing');

  const body: Record<string, unknown> = {
    languageCode: opts.language,
    maxResultCount: 5, // D-07
  };

  if (opts.query) {
    body.textQuery = opts.query;
  }

  if (opts.lat !== undefined && opts.lng !== undefined) {
    body.locationBias = {
      circle: {
        center: { latitude: opts.lat, longitude: opts.lng },
        radius: 50000,
      },
    };
  }

  const res = await fetch('https://places.googleapis.com/v1/places:searchText', {
    method: 'POST',
    headers: {
      'content-type': 'application/json',
      'X-Goog-Api-Key': apiKey,
      'X-Goog-FieldMask': FIELD_MASK,
    },
    body: JSON.stringify(body),
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`Places API failed (${res.status}): ${text}`);
  }

  const data = await res.json() as { places?: unknown[] };
  const list = data?.places ?? [];

  return list.map((raw) => {
    const p = raw as Record<string, any>;
    return {
      google_place_id: String(p.id ?? ''),
      displayName: String(p.displayName?.text ?? ''),
      formattedAddress: p.formattedAddress ? String(p.formattedAddress) : null,
      location: {
        lat: Number(p.location?.latitude ?? 0),
        lng: Number(p.location?.longitude ?? 0),
      },
      primaryType: p.primaryType ? String(p.primaryType) : null,
    };
  }).filter((p) => p.google_place_id.length > 0); // drop malformed
}
```

**B. Create `supabase/functions/resolve-place/index.ts`:**

```typescript
// Edge Function: resolve-place
// POST { query?, lat?, lng?, language? } → { places: ResolvedPlace[] } (max 5)
//
// Authorization: Bearer <user JWT> required (anon allowed at policy level — Function does not RLS-protect itself; it logs cost).
// Cost logging: per-call row in extraction_costs (link_id NULL — Phase 3 Plan 03-01 migration 0005).
//
// Phase 3 D-08 lock: replaces ad-hoc client-side Places calls with a single server-side path
// that keeps Google API key off-device and enforces FieldMask (Phase 2 D-12).

import 'jsr:@supabase/functions-js/edge-runtime.d.ts';
import { createClient } from 'jsr:@supabase/supabase-js@^2';
import { z } from 'npm:zod@^3.23';
import { searchPlaces, type ResolvedPlace } from './pipeline/places-search.ts';

const RequestSchema = z
  .object({
    query: z.string().min(1).max(200).optional(),
    lat: z.number().min(-90).max(90).optional(),
    lng: z.number().min(-180).max(180).optional(),
    language: z.string().min(2).max(8).default('ko'),
  })
  .refine((v) => v.query !== undefined || (v.lat !== undefined && v.lng !== undefined), {
    message: 'Either query or (lat,lng) is required',
  });

function corsHeaders(): HeadersInit {
  return {
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Headers': 'authorization, content-type',
    'Access-Control-Allow-Methods': 'POST, OPTIONS',
  };
}

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response(null, { headers: corsHeaders() });
  if (req.method !== 'POST') {
    return new Response(JSON.stringify({ error: 'method_not_allowed' }), {
      status: 405,
      headers: { 'content-type': 'application/json', ...corsHeaders() },
    });
  }

  const authHeader = req.headers.get('Authorization');
  if (!authHeader?.toLowerCase().startsWith('bearer ')) {
    return new Response(JSON.stringify({ error: 'unauthorized' }), {
      status: 401,
      headers: { 'content-type': 'application/json', ...corsHeaders() },
    });
  }

  const body = await req.json().catch(() => null);
  const parsed = RequestSchema.safeParse(body);
  if (!parsed.success) {
    return new Response(
      JSON.stringify({ error: 'invalid_body', detail: parsed.error.flatten() }),
      { status: 400, headers: { 'content-type': 'application/json', ...corsHeaders() } },
    );
  }

  const supabaseUrl = Deno.env.get('SUPABASE_URL');
  const serviceRole = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY');
  if (!supabaseUrl || !serviceRole) {
    return new Response(JSON.stringify({ error: 'server_misconfigured' }), {
      status: 500,
      headers: { 'content-type': 'application/json', ...corsHeaders() },
    });
  }
  const admin = createClient(supabaseUrl, serviceRole, { auth: { persistSession: false } });

  const t0 = performance.now();
  let places: ResolvedPlace[];
  try {
    places = await searchPlaces(parsed.data);
  } catch (e) {
    console.error('[resolve-place] places api failed:', e);
    return new Response(JSON.stringify({ error: 'places_api_failed' }), {
      status: 502,
      headers: { 'content-type': 'application/json', ...corsHeaders() },
    });
  }
  const duration_ms = Math.round(performance.now() - t0);

  // Phase 2 D-09 pattern: per-call cost row. link_id null = manual search (D-08).
  // cost_usd 0.032 = Text Search Pro tier published rate (Google Places API v1, 2026-Q2).
  await admin.from('extraction_costs').insert({
    link_id: null,
    provider: 'google_places',
    model: 'text-search',
    tokens: null,
    cost_usd: 0.032,
    duration_ms,
  }).then(({ error }) => {
    if (error) console.warn('[resolve-place] cost-log failed:', error.message);
  });

  return new Response(JSON.stringify({ places }), {
    headers: { 'content-type': 'application/json', ...corsHeaders() },
  });
});
```

```typescript
/**
 * Rename a place's display name (name_local). google_place_id remains immutable —
 * users edit the local-language label, not the canonical Google identifier.
 *
 * RLS: `can_edit_board()` SECURITY DEFINER helper gates this UPDATE
 * (existing policy from 0001_init.sql). Non-members get RLS denied.
 *
 * Per Phase 3 D-09: bottom sheet "이름 수정" action.
 */
export async function renamePlace(
  client: MoajoaSupabaseClient,
  id: string,
  newName: string,
): Promise<Place> {
  const trimmed = newName.trim();
  if (trimmed.length === 0) throw new Error('name_local cannot be empty');
  if (trimmed.length > 200) throw new Error('name_local exceeds 200 chars');

  const { data, error } = await client
    .from('places')
    .update({ name_local: trimmed })
    .eq('id', id)
    .select('*')
    .single();
  if (error) throw error;
  return data as Place;
}

/**
 * Soft-delete a place (alias of hidePlace — keeps vote history; row is hidden
 * from board view via `hidden_at is null` filter on lists).
 *
 * Per Phase 3 D-09: bottom sheet "삭제" action.
 *
 * Why alias instead of new function: hidePlace already implements the exact
 * soft-delete semantics. Exporting as `deletePlace` gives the UI an
 * intent-aligned name without duplicating logic (Karpathy §3.2 simplicity).
 */
export const deletePlace = hidePlace;
```
