   ```ts
   import { revalidateTag } from 'next/cache';
   import { timingSafeEqual } from 'node:crypto';
   import { z } from 'zod';
   import { BOARD_REVALIDATE_TAG } from '@/lib/cache';
   import { getRevalidateSecret } from '@/lib/env';

   /**
    * Webhook endpoint called by Supabase Edge Function `extract-youtube`
    * after a `done` broadcast. Invalidates the cached public board so
    * the next visitor gets fresh data (per CONTEXT D-04, D-20).
    *
    * Node runtime is required for `node:crypto.timingSafeEqual` — Edge
    * runtime's WebCrypto lacks this primitive (RESEARCH §Pitfall 9).
    */
   export const runtime = 'nodejs';

   const BodySchema = z.object({
     slug: z.string().min(8).max(32),
     secret: z.string().min(16),
   });

   function safeEqualStr(a: string, b: string): boolean {
     // Length check first — timingSafeEqual throws on length mismatch.
     // Secret length is a known constant (32 bytes hex = 64 chars), so
     // leaking length is acceptable per D-20.
     if (a.length !== b.length) return false;
     return timingSafeEqual(Buffer.from(a, 'utf8'), Buffer.from(b, 'utf8'));
   }

   export async function POST(request: Request) {
     let body: unknown;
     try {
       body = await request.json();
     } catch {
       return Response.json({ ok: false, error: 'invalid json' }, { status: 400 });
     }

     const parsed = BodySchema.safeParse(body);
     if (!parsed.success) {
       return Response.json({ ok: false, error: 'invalid body' }, { status: 400 });
     }

     const secret = getRevalidateSecret();
     if (!secret) {
       return Response.json({ ok: false, error: 'misconfigured' }, { status: 500 });
     }

     if (!safeEqualStr(parsed.data.secret, secret)) {
       return Response.json({ ok: false, error: 'unauthorized' }, { status: 401 });
     }

     revalidateTag(BOARD_REVALIDATE_TAG(parsed.data.slug));

     return Response.json({ ok: true, slug: parsed.data.slug });
   }

   export async function GET() {
     return Response.json({ ok: false, error: 'method not allowed' }, { status: 405 });
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| External (untrusted) → /api/revalidate | Anyone can POST. Defense = secret comparison. |
| Edge Function (Supabase) → web (Vercel) | Same secret on both sides. Defense = REVALIDATE_SECRET env (1Password). |
| Webhook body → revalidateTag(arg) | Slug must be 8-32 chars (zod). Defense against tag-injection. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-02-01 | S (Spoofing) | /api/revalidate POST body.secret | mitigate | timingSafeEqual via node:crypto + zod min(16) (D-20). Length-prefix guard short-circuits length mismatch. |
| T-04-02-02 | T (Tampering) | body.slug propagating to revalidateTag(`board:${slug}`) | mitigate | zod min(8) max(32) caps tag space + revalidateTag is server-only Next.js primitive (no SQL). |
| T-04-02-03 | I (Info disclosure) | revalidate error responses | mitigate | All error responses use minimal `{ ok: false, error: <short> }` JSON — no secret echo, no stack traces. |
| T-04-02-04 | D (Denial of Service) | unauthenticated /api/revalidate flood | accept | v1 has no rate limit (D-19). Per-request work is small (1 tag mark). Phase 6 dogfooding will revisit if abuse observed. |
