Then append, copying the subscribePlanProgress L54-65 skeleton:
```typescript
export type PollEvent =
  | { kind: 'vote'; payload: unknown }
  | { kind: 'comment'; payload: unknown }
  | { kind: 'presence'; viewers: number };

export function subscribePollChannel(
  tripId: string,
  onEvent: (e: PollEvent) => void,
): RealtimeChannel {
  const channel = supabase
    .channel(pollChannelName(tripId))
    .on('broadcast', { event: 'vote' }, (msg) => onEvent({ kind: 'vote', payload: msg.payload }))
    .on('broadcast', { event: 'comment' }, (msg) => onEvent({ kind: 'comment', payload: msg.payload }))
    .on('presence', { event: 'sync' }, () => {
      const state = channel.presenceState();
      onEvent({ kind: 'presence', viewers: Object.keys(state).length });
    })
    .subscribe();
  return channel;
}
```
Caller cleanup is identical: `supabase.removeChannel(channel)` in the useEffect return (the L19-22 leak-guard contract).
