import { PollKeys } from '@moajoa/core';
export function getDeviceToken(): string {
  if (typeof window === 'undefined') return '';            // SSR guard — never run server-side
  let t = window.localStorage.getItem(PollKeys.DeviceToken);
  if (!t) { t = crypto.randomUUID(); window.localStorage.setItem(PollKeys.DeviceToken, t); }
  return t;
}
```

import { createClient } from '@supabase/supabase-js';
import { unstable_cache } from 'next/cache';
import { pollByCode, type Database } from '@moajoa/api';

export const POLL_REVALIDATE_TAG = (code: string): string => `poll:${code}`;

export function getCachedPoll(code: string) {
  return unstable_cache(
    async () => {
      const supabase = createClient<Database>(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
      );
      return pollByCode(supabase, code);   // returns static metadata: {id, trip_id, mode, status, options}
    },
    ['public-poll', code],
    { tags: [POLL_REVALIDATE_TAG(code)], revalidate: 3600 },
  )();
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| anonymous browser → anon RPC | Unauthenticated visitor input (nickname, body, vote payload, device token) crosses to the DB via the anon-granted RPCs. The poll_code is a bearer capability. |
| anonymous viewers ↔ public Realtime channel | Anon-key clients broadcast vote/comment + track presence on a public channel (no JWT, by design). |
| cached SSR render → all anon viewers | The cookies-free cache is shared; any per-viewer state in it poisons everyone. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-19-05 | Input Validation | island sending unvalidated payloads | mitigate | @moajoa/core Zod narrows nickname/availability/body before the call; the DEFINER RPC re-validates server-side (Plan 01). Never sends trip_id/poll_id — only the code (derived server-side). |
| T-19-07 | Information Disclosure | tally render leaking PII | accept | The island renders only getPollTally's shaped output (counts + voter-chosen nicknames; no device tokens — guaranteed by Plan 01). |
| T-19-10 | Tampering | SSR cache poisoning across anon viewers | mitigate | poll-cache caches ONLY static metadata; the island hydrates votes/tally/presence/chat client-side. Verified by acceptance: no tally/vote fetch in page.tsx (Pitfall 2). |
| T-19-02 | Spoofing/Tampering | vote stuffing via rotated device tokens | accept (bounded) | Device token is a dedup handle, not a security boundary (Pitfall 1). Non-guessable code + poll-open gate + dedup index bound the blast radius; hard IP limiting deferred (A4). |
| T-19-03 | DoS | comment flooding | mitigate | Server-side length cap + 5s per-device throttle + poll-open gate (Plan 01). Client adds optimistic UI only. |
| T-19-05b | Elevation of Privacy | private Realtime channel requiring JWT | mitigate (design) | Uses a PUBLIC channel (anon visitors have no JWT) — matches the extract:/plan: idiom. Never `config:{private:true}` on the anon island. |
</threat_model>
