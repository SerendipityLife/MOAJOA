   * Create the date poll for an EXISTING trip (web share-sheet dates/both flow,
   * 25-07 Gap 1). Direct insert gated by the existing date_polls_write RLS
   * (`can_edit_trip(trip_id)`, 0018) — no new RPC/migration. poll_code is
   * generated by the ensure_poll_code before-insert trigger and returned here.
   */
  export async function createDatePoll(
    client: MoajoaSupabaseClient,
    tripId: string,
    mode: 'range' | 'grid' = 'range',
  ): Promise<{ id: string; poll_code: string | null; mode: 'range' | 'grid'; status: 'open' | 'closed' }> {
    const { data, error } = await client
      .from('date_polls')
      .insert({ trip_id: tripId, mode })
      .select('id, poll_code, mode, status')
      .single();
    if (error) throw error;
    return data as { id: string; poll_code: string | null; mode: 'range' | 'grid'; status: 'open' | 'closed' };
  }
