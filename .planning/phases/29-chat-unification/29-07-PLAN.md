       create or replace function public_trip_messages(p_slug text)

         limit 1;

         select coalesce(jsonb_agg(

             'nickname', tm.nickname,
             'body', tm.body,
             'reply_to_place_id', tm.reply_to_place_id,
             'created_at', tm.created_at

         into v_messages
         from (
